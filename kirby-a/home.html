// 기본 js함수 60fps 그러나 모니터 프레임리셋수에 따라 다름. 1초에 144번 실행될 수도 그런경우 1초에 144px 이동하는 것임.
function 프레임마다실행() {
  requestAnimationFrame(프레임마다실행);

  //프레임마다 canvas 리셋하여 잔상 삭제
  ctx.clearRect(0,0, canvas.width, canvas.height);

  dino.x++;
  dino.draw();

}

프레임마다실행();



//장애물 여러개 관리 → 스폰 후 array에 담기
var timer = 0;
var cacti = []; // 장애물 여러개의 array


function 프레임마다실행() {
  requestAnimationFrame(프레임마다실행);
  timer++;

  ctx.clearRect(0,0, canvas.width, canvas.height);

    var cactus = new Cactus();
    cactus.draw();
    cacti.push(cactus);
  }

  cacti.forEach((a)=>{
    a.x--;
    a.draw();
  })

  dino.draw();

}

프레임마다실행();
