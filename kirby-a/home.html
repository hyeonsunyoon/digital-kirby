let keys = {
 left: false,
 right: false
}


// 키 입력감지 조건문
document.addEventListener("keydown", (e) => {
 if (e.key === "ArrowLeft") {
  keys.left = true;
 }
 if (e.key === "ArrowRight") {
  keys.right = true;
 }
})

document.addEventListener("keyup", (e) => {
 if (e.key === "ArrowLeft") {
  keys.left = false;
 }
 if (e.key === "ArrowRight") {
  keys.right = false;
 }
})


// 플레이어 이동함수
function movePlayer() {
 const playerLeft = parseFloat(getComputedStyle(player).left);
 let speed = 3;

 if (keys.left && playerLeft > 0) {
  player.style.left = `${playerLeft - speed}px`;
 }

 if (keys.right && playerLeft < window.innerWidth - player.offsetWidth) {
  player.style.left = `${playerLeft + speed}px`;
 }
}


// 게임 루프
function update() {
 movePlayer();
 requestAnimationFrame(update);
}

// 게임 실행
update();
