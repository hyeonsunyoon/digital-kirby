const canvas = document.getElementById("gameCanvas");
const ctx = canvas.getContext("2d");

const playerImg = new Image();
playerImg.src = "player.png";
const enemyImg = new Image();
enemyImg.src = "raindrop.png";
const pointImg = new Image();
pointImg.src = "umbrella.png";

const player = {
    x: 100,
    y: canvas.height - 40,
    width: 80,
    height: 40,
    speed: 3,
    color: "blue"
};

let enemies = [];
let points = [];
let score = 500;
let stage = 1;
let comboCount = 0;
let comboTimer = null;

let keys = {
    left: false,
    right: false,
    space: false,
};

document.addEventListener("keydown", (e) => {
    if (e.key === "ArrowLeft") keys.left = true;
    if (e.key === "ArrowRight") keys.right = true;
    if (e.key === " ") keys.space = true;
});

document.addEventListener("keyup", (e) => {
    if (e.key === "ArrowLeft") keys.left = false;
    if (e.key === "ArrowRight") keys.right = false;
    if (e.key === " ") keys.space = false;
});

// 충돌 감지
function isColliding(a, b) {
    return !(
        a.x > b.x + b.width ||
        a.x + a.width < b.x ||
        a.y > b.y + b.height ||
        a.y + a.height < b.y
    );
}

// 적 생성
function spawnEnemy() {
    enemies.push({
        x: Math.random() * (canvas.width - 40),
        y: -40,
        width: 15,
        height: 36,
        speed: 2 * stage
    });
    setTimeout(spawnEnemy, Math.random() / (stage * stage) + 100);
}

// 아이템 생성
function spawnPoint() {
    points.push({
        x: Math.random() * (canvas.width - 20),
        y: -20,
        width: 51,
        height: 42,
        speed: 1
    });
    setTimeout(spawnPoint, Math.random() * 10000);
}

function drawPlayer() {
    ctx.drawImage(playerImg, player.x, player.y, player.width, player.height);
}

function drawObjects(objects, image) {
    for (let obj of objects) {
    	ctx.drawImage(image, obj.x, obj.y, obj.width, obj.height);
    }
}

function updateObjects() {
    enemies.forEach((e, i) => {
        e.y += e.speed;
        if (isColliding(player, e)) {
            score = Math.max(0, score - (1 + stage));
            comboCount = 0;
            enemies.splice(i, 1);
        } else if (e.y > canvas.height) {
            enemies.splice(i, 1);
        }
    });

    points.forEach((p, i) => {
        p.y += p.speed;
        if (isColliding(player, p)) {
            comboCount++;
            score += (10 * stage) + (comboCount - 1);
            clearTimeout(comboTimer);
            comboTimer = setTimeout(() => comboCount = 0, 3000);
            points.splice(i, 1);
        } else if (p.y > canvas.height) {
            points.splice(i, 1);
        }
    });
}

function movePlayer() {
    let speed = keys.space ? 6 : 3;
    if (keys.left) player.x -= speed;
    if (keys.right) player.x += speed;

    // 경계 제한
    player.x = Math.max(0, Math.min(canvas.width - player.width, player.x));
}

function drawScore() {
    ctx.fillStyle = "black";
    ctx.font = "20px monospace";
    ctx.fillText(`HP: ${score}`, 10, 30);
}

function gameLoop() {
    ctx.clearRect(0, 0, canvas.width, canvas.height);
    movePlayer();
    updateObjects();
    drawPlayer();
    drawObjects(enemies,enemyImg);
    drawObjects(points,pointImg);
    drawScore();
    requestAnimationFrame(gameLoop);
}

spawnEnemy();
spawnPoint();
setInterval(() => stage++, 15000);
gameLoop();
