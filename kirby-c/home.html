    // 이미지 목록 + 설명
    const imagePairs = [
      {
        original: 'original1.jpeg',
        restored: 'restored1.png',
        description: "Karl Blossfeldt\nKarl Blossfeldt's Urformen der Kunst (1928)"
      },
      {
        original: 'original2.jpeg',
        restored: 'restored2.png',
        description: "Karl Blossfeldt\nKarl Blossfeldt's Urformen der Kunst (1928)"
      },
      {
        original: 'original3.jpeg',
        restored: 'restored3.png',
        description: "Karl Blossfeldt\nKarl Blossfeldt's Urformen der Kunst (1928)"
      }
    ];

    let index = 0;

    const container = document.getElementById('imageContainer');
    const originalImg = container.querySelector('.original');
    const restoredImg = container.querySelector('.restored');
    const typewriterElement = document.getElementById('typewriter');

    const prevBtn = document.getElementById('prevBtn');
    const nextBtn = document.getElementById('nextBtn');

    // 타자기 효과 함수
    function typeWriter(text) {
      typewriterElement.textContent = '';
      let i = 0;
      function type() {
        if (i < text.length) {
          typewriterElement.textContent += text.charAt(i);
          i++;
          setTimeout(type, 40);
        }
      }
      type();
    }

    // 이미지 업데이트
    function updateImage() {
      const { original, restored, description } = imagePairs[index];
      originalImg.src = original;
      restoredImg.src = restored;
      typeWriter(description);
    }

    // 초기 설정
    updateImage();

    // Hover 이벤트
    container.addEventListener('mouseenter', () => {
      container.classList.add('hovered');
    });

    container.addEventListener('mouseleave', () => {
      container.classList.remove('hovered');
    });

    // 클릭 시 다음 이미지
    container.addEventListener('click', () => {
      index = (index + 1) % imagePairs.length;
      updateImage();
    });

    // 화살표 버튼
    prevBtn.addEventListener('click', (e) => {
      e.stopPropagation(); // 클릭 이벤트 막기
      index = (index - 1 + imagePairs.length) % imagePairs.length;
      updateImage();
    });

    nextBtn.addEventListener('click', (e) => {
      e.stopPropagation();
      index = (index + 1) % imagePairs.length;
      updateImage();
    });
